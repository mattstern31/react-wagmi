# React

Plugin for generating type-safe React Hooks from configuration `contracts`. Requires `wagmi@>=0.11`.

```ts
import { react } from '@wagmi/cli/plugins'
```

## Usage

```ts
import { defineConfig } from '@wagmi/cli'
import { react } from '@wagmi/cli/plugins'

export default defineConfig({
  plugins: [
    react({
      useContractRead: true,
      useContractFunctionRead: true,
    }),
  ],
})
```

## Configuration

### useContract (optional)

Generate [`useContract`](/react/hooks/useContract) hook. Defaults to `true`.

```ts {7}
import { defineConfig } from '@wagmi/cli'
import { react } from '@wagmi/cli/plugins'

export default defineConfig({
  plugins: [
    react({
      useContract: true,
    }),
  ],
})
```

### useContractEvent (optional)

Generate [`useContractEvent`](/react/hooks/useContractEvent) hook. Defaults to `true`.

```ts {7}
import { defineConfig } from '@wagmi/cli'
import { react } from '@wagmi/cli/plugins'

export default defineConfig({
  plugins: [
    react({
      useContractEvent: true,
    }),
  ],
})
```

### useContractItemEvent (optional)

Generate [`useContractEvent`](/react/hooks/useContractEvent) hook for each event in ABI. Defaults to `true`.

```ts {7}
import { defineConfig } from '@wagmi/cli'
import { react } from '@wagmi/cli/plugins'

export default defineConfig({
  plugins: [
    react({
      useContractItemEvent: true,
    }),
  ],
})
```
